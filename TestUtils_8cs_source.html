97 internal static string ReadFirstNodeValue(Stream stream, string nodeName)
98 {
99 if (stream == null)
100 {
101 throw new ArgumentNullException(nameof(stream));
102 }
103 if (string.IsNullOrEmpty(nodeName))
104 {
105 throw new ArgumentException("Node name must be specified.", nameof(nodeName));
106 }
107 using (var reader = XmlReader.Create(stream, new XmlReaderSettings
108 {
109 IgnoreComments = true,
110 IgnoreWhitespace = true,
111 DtdProcessing = DtdProcessing.Ignore
112 }))
113 {
114 while (reader.Read())
115 {
116 if (reader.NodeType == XmlNodeType.Element && reader.LocalName == nodeName)
117 {
118 // ReadElementContentAsString moves past the end element automatically
119 return reader.ReadElementContentAsString();
120 }
121 }
122 }
123
124 return null; // not found
125 }
